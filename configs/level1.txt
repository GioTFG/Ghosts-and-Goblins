# This is an example of a well-formed config file for the game.
# It recreates the first part of the first official level of Ghost 'n Goblins.
# It is a sort of fusion between JSON and CSV.
# Every entry has its own line in the file, and is made of two parts: the option, and the value(s).
# Every option has a specific number and type of value that it can be assigned to.
# Some options allow multiple values. These MUST be introduced by an open square bracket and closed by its closed equivalent.
# The brackets must be used even if there's only one entry for the value.
# Lines starting with "#" and empty lines will be ignored.

# This file will contain comments that describe what each of the entries do.

# Hero_Start_Pos: <int, int> --- Arthur's starting position
Hero_Start_Pos: 94, 152

# Size: <int>, <int> --- It represents the size of the arena.
Size: 3584, 240

# Lives: <int> --- The total lives Arthur starts with.
Lives: 3

# Enemies: <list<Enemy>> --- A list of all the enemies that statically spawn in the level
Enemies: [
    # Each enemy must be in the form of: <EnemyClass>: <Initial Position>, <ClassArguments>
    # <InitialPosition> := <int>, <int>
    # <ClassArguments> depends on the specific enemy...
    # - Plant has none
    # - Zombie has direction (Left or Right)
    Plant: 1020, 90

    Zombie: 242, 192, Right
]
